### 2.4 版本管理规范

**语义化版本（SemVer）**：
```
主版本。次版本。修订号
  ↑      ↑      ↑
  |      |      └─ 向后兼容的 Bug 修复
  |      └─ 向后兼容的功能新增
  └─ 不向后兼容的重大变更

示例:
1.0.0  # 初始稳定版本
1.0.1  # Bug 修复
1.1.0  # 新增功能
2.0.0  # 破坏性变更
```

**版本兼容性**：
```yaml
兼容:
  - ^1.0.0  # >=1.0.0 且 <2.0.0
  - ~1.2.0  # >=1.2.0 且 <1.3.0
  - >=1.0.0 # 1.0.0及以上
  - *       # 任意版本

不兼容:
  - <1.0.0  # 1.0.0 以下
  - =1.0.0  # 精确匹配
```

## 3. 技能分发机制

### 3.1 安装器实现

```typescript
// skill_installer.ts

import axios from 'axios';
import { exec } from 'child_process';
import { promisify } from 'util';
import fs from 'fs-extra';
import path from 'path';
import semver from 'semver';

const execAsync = promisify(exec);

interface SkillPackage {
  name: string;
  version: string;
  source: 'clawhub' | 'github';
  downloadUrl: string;
  metadata: any;
}

interface InstallOptions {
  targetDir: string;
  skipDependencies?: boolean;
  force?: boolean;
}

export class SkillInstaller {
  private clawhubApi = 'https://api.clawhub.ai/v1';
  
  /**
   * 安装技能
   */
  async install(skillName: string, options: InstallOptions): Promise<void> {
    console.log(`正在安装技能：${skillName}`);
    
    // 1. 获取技能信息
    const skillPackage = await this.fetchSkillInfo(skillName);
    
    // 2. 检查版本兼容性
    await this.checkCompatibility(skillPackage);
    
    // 3. 下载技能包
    const tempDir = await this.downloadSkill(skillPackage);
    
    // 4. 安装依赖
    if (!options.skipDependencies) {
      await this.installDependencies(skillPackage.metadata.dependencies);
    }
    
    // 5. 移动到目标目录
    const targetPath = path.join(options.targetDir, skillName);
    
    if (await fs.pathExists(targetPath) && !options.force) {
      throw new Error(`技能 ${skillName} 已存在，使用 --force 覆盖`);
    }
    
    await fs.move(tempDir, targetPath, { overwrite: true });
    
    // 6. 更新技能索引
    await this.updateSkillIndex(skillPackage.metadata);
    
    console.log(`✅ 技能 ${skillName}@${skillPackage.version} 安装成功`);
  }

  /**
   * 获取技能信息
   */
  private async fetchSkillInfo(skillName: string): Promise<SkillPackage> {
    // 优先从 ClawHub 获取
    try {
      const response = await axios.get(`${this.clawhubApi}/skills/${skillName}`);
      return {
        name: response.data.name,
        version: response.data.version,
        source: 'clawhub',
        downloadUrl: response.data.downloadUrl,
        metadata: response.data
      };
    } catch (error) {
      // ClawHub 不存在，尝试 GitHub
      console.log('ClawHub 未找到，尝试 GitHub...');
      return this.fetchFromGitHub(skillName);
    }
  }

  /**
   * 从 GitHub 获取技能
   */
  private async fetchFromGitHub(skillName: string): Promise<SkillPackage> {
    // 解析技能名（格式：author/repo）
    const [owner, repo] = skillName.split('/');
    
    const response = await axios.get(
      `https://api.github.com/repos/${owner}/${repo}/releases/latest`
    );
    
    return {
      name: repo,
      version: response.data.tag_name,
      source: 'github',
      downloadUrl: response.data.zipball_url,
      metadata: {
        name: repo,
        version: response.data.tag_name,
        description: response.data.name
      }
    };
  }

  /**
   * 检查兼容性
   */
  private async checkCompatibility(skillPackage: SkillPackage): Promise<void> {
    const { metadata } = skillPackage;
    
    // 检查 OpenCode 版本
    const opencodeVersion = await this.getOpenCodeVersion();
    if (metadata.min_opencode_version &&
        !semver.gte(opencodeVersion, metadata.min_opencode_version)) {
      throw new Error(
        `技能需要 OpenCode >= ${metadata.min_opencode_version}，当前版本 ${opencodeVersion}`
      );
    }
    
    // 检查依赖技能
    if (metadata.dependencies) {
      for (const [depName, versionRange] of Object.entries(metadata.dependencies)) {
        const installed = await this.getInstalledSkillVersion(depName);
        if (!installed || !semver.satisfies(installed, versionRange)) {
          throw new Error(
            `依赖技能 ${depName}@${versionRange} 未安装或版本不兼容`
          );
        }
      }
    }
  }

  /**
   * 下载技能包
   */
  private async downloadSkill(skillPackage: SkillPackage): Promise<string> {
    const tempDir = path.join('/tmp', `skill-${Date.now()}`);
    await fs.ensureDir(tempDir);
    
    const response = await axios.get(skillPackage.downloadUrl, {
      responseType: 'arraybuffer'
    });
    
    const zipPath = path.join(tempDir, 'skill.zip');
    await fs.writeFile(zipPath, response.data);
    
    // 解压
    await execAsync(`unzip -o ${zipPath} -d ${tempDir}`);
    
    return tempDir;
  }

  /**
   * 安装依赖技能
   */
  private async installDependencies(dependencies: Record<string, string>): Promise<void> {
    for (const [depName, versionRange] of Object.entries(dependencies)) {
      console.log(`安装依赖：${depName}@${versionRange}`);
      
      // 递归安装
      await this.install(depName, {
        targetDir: path.dirname(process.cwd()),
        skipDependencies: false,
        force: false
      });
    }
  }

  /**
   * 更新技能索引
   */
  private async updateSkillIndex(metadata: any): Promise<void> {
    // 更新 SKILLS.md 或 skill_index.json
    // 实现略
  }

  private async getOpenCodeVersion(): Promise<string> {
    // 获取当前 OpenCode 版本
    return '1.0.0';
  }

  private async getInstalledSkillVersion(skillName: string): Promise<string | null> {
    // 获取已安装技能版本
    return null;
  }
}
```

### 3.2 一键安装命令

```typescript
// commands/skills_install.ts

import { Command } from 'commander';
import { SkillInstaller } from '../skill_installer';

const program = new Command();

program
  .command('skills:install <skill-name>')
  .alias('skills:i')
  .description('安装技能')
  .option('-s, --source <source>', '指定源 (clawhub|github)', 'clawhub')
  .option('--skip-deps', '跳过依赖安装', false)
  .option('-f, --force', '覆盖已存在的技能', false)
  .option('-v, --verbose', '详细输出', false)
  .action(async (skillName, options) => {
    const installer = new SkillInstaller();
    
    try {
      await installer.install(skillName, {
        targetDir: process.cwd(),
        skipDependencies: options.skipDeps,
        force: options.force
      });
      
      console.log('✅ 安装成功');
    } catch (error: any) {
      console.error('❌ 安装失败:', error.message);
      if (options.verbose) {
        console.error(error.stack);
      }
      process.exit(1);
    }
  });

// 使用示例
// opencode skills:install code-reviewer
// opencode skills:i code-reviewer --skip-deps
```

```typescript
interface ResolutionResult {

export class DependencyResolver {
  /**
   * 解析依赖树
   */
  async resolve(skillName: string, versionRange: string): Promise<ResolutionResult[]> {
    const resolutionQueue: Array<{name: string, version: string}> = [];
    const resolved: Map<string, string> = new Map();
    const result: ResolutionResult[] = [];
    
    // 初始技能入队
    resolutionQueue.push({ name: skillName, version: versionRange });
    
    while (resolutionQueue.length > 0) {
      const { name, version } = resolutionQueue.shift()!;
      
      // 跳过已解析的
      if (resolved.has(name)) {
        // 检查版本冲突
        if (!semver.satisfies(resolved.get(name)!, version)) {
          throw new Error(
            `版本冲突：${name} 已安装 ${resolved.get(name)}，需要 ${version}`
          );
        }
        continue;
      }
      
      // 获取技能信息
      const skillInfo = await this.fetchSkillInfo(name, version);
      resolved.set(name, skillInfo.version);
      
      result.push({
        skill: name,
        version: skillInfo.version,
        dependencies: skillInfo.dependencies || []
      });
      
      // 依赖入队
      for (const dep of skillInfo.dependencies || []) {
        resolutionQueue.push(dep);
      }
    }
```

```typescript
// rating_system.ts

interface SkillRating {
  skill_name: string;
  average: number;      // 平均分 (0-5)
  count: number;        // 评分数量
  distribution: {
    5: number;
    4: number;
    3: number;
    2: number;
    1: number;
  };
}

export class RatingSystem {
  private redis: Redis;

  constructor(redis: Redis) {
    this.redis = redis;
  }

  /**
   * 添加评分
   */
  async addRating(skillName: string, userId: string, rating: number): Promise<void> {
    if (rating < 1 || rating > 5) {
      throw new Error('评分必须在 1-5 之间');
    }

    const key = `rating:${skillName}`;
    const userKey = `rating:${skillName}:${userId}`;
    
    // 检查用户是否已评分
    const existingRating = await this.redis.get(userKey);
    
    if (existingRating) {
      // 更新评分
      await this.redis.zrem(key, userId);
    }
    
    // 添加新评分
    await this.redis.zadd(key, rating, userId);
    await this.redis.set(userKey, rating);
    
    // 更新统计
    await this.updateStats(skillName);
  }

  /**
   * 获取评分统计
   */
  async getRating(skillName: string): Promise<SkillRating> {
    const key = `rating:${skillName}`;
    
    const ratings = await this.redis.zrange(key, 0, -1, 'WITHSCORES');
    const count = ratings.length / 2;
    
    if (count === 0) {
      return {
        skill_name: skillName,
        average: 0,
        count: 0,
        distribution: { 5: 0, 4: 0, 3: 0, 2: 0, 1: 0 }
      };
    }
    
    // 计算平均分和分布
    let sum = 0;
    const distribution = { 5: 0, 4: 0, 3: 0, 2: 0, 1: 0 };
    
    for (let i = 0; i < ratings.length; i += 2) {
      const score = parseFloat(ratings[i + 1]);
      sum += score;
      distribution[score as keyof typeof distribution]++;
    }
    
    return {
      skill_name: skillName,
      average: sum / count,
      count,
      distribution
    };
  }

  private async updateStats(skillName: string): Promise<void> {
    const stats = await this.getRating(skillName);
    await this.redis.setex(
      `rating:stats:${skillName}`,
      3600,
      JSON.stringify(stats)
    );
  }
}
```

```typescript
// download_tracker.ts

export class DownloadTracker {
  private redis: Redis;

  constructor(redis: Redis) {
    this.redis = redis;
  }

  /**
   * 记录下载
   */
  async trackDownload(skillName: string, userId: string): Promise<void> {
    const date = new Date().toISOString().split('T')[0]; // YYYY-MM-DD
    
    // 总下载量
    await this.redis.incr(`download:total:${skillName}`);
    
    // 每日下载量
    await this.redis.incr(`download:daily:${skillName}:${date}`);
    
    // 用户下载记录（防止重复计数）
    const userKey = `download:user:${skillName}:${userId}`;
    const hasDownloaded = await this.redis.get(userKey);
    
    if (!hasDownloaded) {
      await this.redis.setex(userKey, 86400 * 7, '1'); // 7 天内不重复计数
      await this.redis.incr(`download:unique:${skillName}`);
    }
  }

  /**
   * 获取下载统计
   */
  async getStats(skillName: string): Promise<any> {
    const [total, unique, daily] = await Promise.all([
      this.redis.get(`download:total:${skillName}`),
      this.redis.get(`download:unique:${skillName}`),
      this.redis.get(`download:daily:${skillName}:${new Date().toISOString().split('T')[0]}`)
    ]);
    
    return {
      total: parseInt(total || '0'),
      unique: parseInt(unique || '0'),
      daily: parseInt(daily || '0')
    };
  }

  /**
   * 获取热门技能排行榜
   */
  async getTopSkills(limit: number = 10): Promise<any[]> {
    // 获取所有技能的下载量
    const pattern = 'download:total:*';
    const keys = await this.redis.keys(pattern);
    
    const skills: Array<{name: string, downloads: number}> = [];
    
    for (const key of keys) {
      const name = key.replace('download:total:', '');
      const downloads = await this.redis.get(key);
      skills.push({ name, downloads: parseInt(downloads || '0') });
    }
    
    // 排序并返回 Top N
    return skills
      .sort((a, b) => b.downloads - a.downloads)
      .slice(0, limit);
  }
}
```

```typescript
// security_scanner.ts

interface SecurityIssue {
  severity: 'critical' | 'high' | 'medium' | 'low';
  type: string;
  description: string;
  file: string;
  line?: number;
  suggestion: string;
}

interface ScanResult {
  skill_name: string;
  scan_date: string;
  passed: boolean;
  issues: SecurityIssue[];
  score: number; // 0-100
}

export class SecurityScanner {
  private checks: Array<(code: string, file: string) => SecurityIssue[]> = [];

  constructor() {
    this.initializeChecks();
  }

  private initializeChecks(): void {
    // 60+ 项安全检查
    this.checks.push(this.checkDangerousCommands);
    this.checks.push(this.checkAPIKeyHardcoding);
    this.checks.push(this.checkSQLInjection);
    this.checks.push(this.checkXSS);
    this.checks.push(this.checkPathTraversal);
    this.checks.push(this.checkInsecureRandom);
    // ... 更多检查
  }

  /**
   * 执行安全扫描
   */
  async scan(skillPath: string): Promise<ScanResult> {
    const issues: SecurityIssue[] = [];
    
    // 遍历技能目录
    const files = await this.getAllFiles(skillPath);
    
    for (const file of files) {
      const content = await fs.readFile(file, 'utf-8');
      
      // 执行所有检查
      for (const check of this.checks) {
        const fileIssues = check(content, file);
        issues.push(...fileIssues);
      }
    }
    
    // 计算安全评分
    const score = this.calculateScore(issues);
    
    return {
      skill_name: path.basename(skillPath),
      scan_date: new Date().toISOString(),
      passed: score >= 60,
      issues,
      score
    };
  }

  /**
   * 检查危险命令
   */
  private checkDangerousCommands(code: string, file: string): SecurityIssue[] {
    const issues: SecurityIssue[] = [];
    
    const dangerousPatterns = [
      { pattern: /exec\s*\(\s*['"`].*rm\s+-rf/, desc: '使用 rm -rf 命令' },
      { pattern: /execSync\s*\(\s*['"`].*chmod\s+777/, desc: '设置 777 权限' },
      { pattern: /eval\s*\(/, desc: '使用 eval()' },
      { pattern: /Function\s*\(/, desc: '使用 Function 构造函数' },
      { pattern: /child_process\.exec/, desc: '执行系统命令' }
    ];
    
    for (const { pattern, desc } of dangerousPatterns) {
      const matches = code.match(pattern);
      if (matches) {
        issues.push({
          severity: 'high',
          type: 'dangerous_command',
          description: desc,
          file,
          suggestion: '避免使用危险命令，使用安全的替代方案'
        });
      }
    }
    
    return issues;
  }

  /**
   * 检查 API 密钥硬编码
   */
  private checkAPIKeyHardcoding(code: string, file: string): SecurityIssue[] {
    const issues: SecurityIssue[] = [];
    
    const apiKeyPatterns = [
      /api[_-]?key\s*[:=]\s*['"][a-zA-Z0-9]{20,}['"]/i,
      /secret\s*[:=]\s*['"][a-zA-Z0-9]{20,}['"]/i,
      /password\s*[:=]\s*['"][^'"]+['"]/i
    ];
    
    for (const pattern of apiKeyPatterns) {
      const matches = code.match(pattern);
      if (matches) {
        issues.push({
          severity: 'critical',
          type: 'hardcoded_secret',
          description: '检测到硬编码的密钥/密码',
          file,
          suggestion: '使用环境变量或密钥管理服务'
        });
      }
    }
    
    return issues;
  }

  /**
   * 检查 SQL 注入
   */
  private checkSQLInjection(code: string, file: string): SecurityIssue[] {
    const issues: SecurityIssue[] = [];
    
    const sqlPatterns = [
      /execute\s*\(\s*['"]SELECT.*\+/,
      /query\s*\(\s*['"]INSERT.*\+/,
      /exec\s*\(\s*['"]UPDATE.*\+/
    ];
    
    for (const pattern of sqlPatterns) {
      const matches = code.match(pattern);
      if (matches) {
        issues.push({
          severity: 'high',
          type: 'sql_injection',
          description: '可能存在 SQL 注入风险',
          file,
          suggestion: '使用参数化查询'
        });
      }
    }
    
    return issues;
  }

  private calculateScore(issues: SecurityIssue[]): number {
    let score = 100;
    
    for (const issue of issues) {
      switch (issue.severity) {
        case 'critical':
          score -= 25;
          break;
        case 'high':
          score -= 15;
          break;
        case 'medium':
          score -= 5;
          break;
        case 'low':
          score -= 2;
          break;
      }
    }
    
    return Math.max(0, score);
  }

  private async getAllFiles(dir: string): Promise<string[]> {
    const files: string[] = [];
    const items = await fs.readdir(dir);
    
    for (const item of items) {
      const fullPath = path.join(dir, item);
      const stat = await fs.stat(fullPath);
      
      if (stat.isDirectory()) {
        files.push(...await this.getAllFiles(fullPath));
      } else if (stat.isFile() && /\.(py|js|ts|sh)$/.test(item)) {
        files.push(fullPath);
      }
    }
    
    return files;
  }
}
```
