## 1. 技能三层加载架构

### 1.1 架构设计原理

OpenClaw 采用三层技能加载架构，灵感来源于 Unix/Linux 系统的配置文件层级设计（`/etc` → `~/.config` → 项目级）。这种设计实现了：

- **关注点分离**：系统技能、个人技能、项目技能各司其职
- **优先级隔离**：高优先级技能覆盖低优先级，支持项目定制
- **灵活扩展**：每层可独立管理，互不干扰

```
┌─────────────────────────────────────────────────────────────────────────┐
│                    OpenClaw 技能三层加载架构                              │
├─────────────────────────────────────────────────────────────────────────┤
│                                                                         │
│  ┌─────────────────────────────────────────────────────────────────┐   │
│  │  Layer 1: 工作区技能 (Workspace Skills)                          │   │
│  │  ─────────────────────────────────────────                       │   │
│  │  路径：<workspace>/.opencode/skills/                             │   │
│  │  优先级：最高 (Priority = 3)                                     │   │
│  │  作用域：当前项目独占                                            │   │
│  │  加载时机：项目启动时                                            │   │
│  │  典型用途：项目特定技能、团队定制技能                            │   │
│  │  示例：/Users/myproject/.opencode/skills/code-review/           │   │
│  └─────────────────────────────────────────────────────────────────┘   │
│                                    ▲                                    │
│                                    │ 优先级覆盖                          │
│  ┌─────────────────────────────────────────────────────────────────┐   │
│  │  Layer 2: 本地技能 (Local Skills)                                │   │
│  │  ─────────────────────────────────────────                       │   │
│  │  路径：~/.config/opencode/skills/                                │   │
│  │  优先级：中等 (Priority = 2)                                     │   │
│  │  作用域：当前用户所有项目                                        │   │
│  │  加载时机：用户会话初始化                                        │   │
│  │  典型用途：个人常用技能、开发习惯定制                            │   │
│  │  示例：~/.config/opencode/skills/file-manager/                  │   │
│  └─────────────────────────────────────────────────────────────────┘   │
│                                    ▲                                    │
│                                    │ 优先级覆盖                          │

│  │  Layer 3: 全局技能 (Global Skills)                               │   │

│  │  路径：/usr/local/share/opencode/skills/ 或系统预定义路径         │   │
│  │  优先级：最低 (Priority = 1)                                     │   │
│  │  作用域：所有用户                                                │   │
│  │  加载时机：系统启动时                                            │   │
│  │  典型用途：基础技能、系统预装技能                                │   │
│  │  示例：/usr/local/share/opencode/skills/web-search/             │   │
│  └─────────────────────────────────────────────────────────────────┘   │
```

### 2.2 递归目录扫描算法

**Python 实现**：
```python
import os
import json
from pathlib import Path
from typing import Dict, List, Optional
from dataclasses import dataclass
from datetime import datetime

@dataclass
class SkillMetadata:
    """技能元数据"""
    name: str
    version: str
    description: str
    author: str
    created: datetime
    updated: datetime
    tags: List[str]
    category: str
    triggers: List[str]
    path: str
    layer: str  # workspace/local/global
    priority: int
    mtime: float  # 修改时间戳

class SkillScanner:
    """技能扫描器"""
    
    SKILL_FILES = ['SKILL.md', 'metadata.json']
    
    def __init__(self):
        self.scanned_skills: Dict[str, SkillMetadata] = {}
    
    def scan_directory(self, directory: str, layer: str, priority: int) -> List[SkillMetadata]:
        """
        递归扫描目录中的所有技能
        
        Args:
            directory: 技能根目录
            layer: 层级标识 (workspace/local/global)
            priority: 优先级 (1-3)
        
        Returns:
            技能元数据列表
        """
        skills = []
        root_path = Path(directory)
        
        if not root_path.exists():
            return skills
        
        # 遍历所有子目录
        for skill_dir in root_path.iterdir():
            if not skill_dir.is_dir():
                continue
            
            # 跳过隐藏目录和特殊目录
            if skill_dir.name.startswith('.') or skill_dir.name in ['node_modules', '__pycache__']:
                continue
            
            # 检查是否为有效技能目录
            skill_metadata = self._scan_skill(skill_dir, layer, priority)
            if skill_metadata:
                skills.append(skill_metadata)
        
        return skills
    
    def _scan_skill(self, skill_dir: Path, layer: str, priority: int) -> Optional[SkillMetadata]:
        """
        扫描单个技能目录
        
        Args:
            skill_dir: 技能目录路径
            layer: 层级标识
            priority: 优先级
        
        Returns:
            技能元数据，如果无效则返回 None
        """
        # 检查 SKILL.md 是否存在
        skill_md_path = skill_dir / 'SKILL.md'
        metadata_json_path = skill_dir / 'metadata.json'
        
        if not skill_md_path.exists() and not metadata_json_path.exists():
            return None  # 不是有效技能目录
        
        # 优先读取 metadata.json（更快）
        if metadata_json_path.exists():
            try:
                with open(metadata_json_path, 'r', encoding='utf-8') as f:
                    metadata = json.load(f)
                return self._build_skill_metadata(metadata, skill_dir, layer, priority)
            except Exception as e:
                print(f"解析 metadata.json 失败：{skill_dir} - {e}")
        
        # 回退到解析 SKILL.md
        if skill_md_path.exists():
            try:
                metadata = self._parse_skill_md(skill_md_path)
                return self._build_skill_metadata(metadata, skill_dir, layer, priority)
            except Exception as e:
                print(f"解析 SKILL.md 失败：{skill_dir} - {e}")
        
        return None
    
    def _parse_skill_md(self, skill_md_path: Path) -> Dict:
        """
        解析 SKILL.md 文件提取元数据
        
        解析 YAML frontmatter 格式：
        ---
        name: skill-name
        version: 1.0.0
        ...
        ---
        """
        import re
        
        with open(skill_md_path, 'r', encoding='utf-8') as f:
            content = f.read()
        
        # 提取 YAML frontmatter
        frontmatter_match = re.match(r'^---\s*\n(.*?)\n---\s*\n', content, re.DOTALL)
        if not frontmatter_match:
            raise ValueError(f"SKILL.md 缺少 frontmatter: {skill_md_path}")
        
        frontmatter = frontmatter_match.group(1)
        metadata = {}
        
        # 简单 YAML 解析
        for line in frontmatter.split('\n'):
            if ':' in line:
                key, value = line.split(':', 1)
                key = key.strip()
                value = value.strip()
                
                # 处理列表格式 [tag1, tag2]
                if value.startswith('[') and value.endswith(']'):
                    value = [v.strip() for v in value[1:-1].split(',')]
                
                metadata[key] = value
        
        # 补充默认值
        metadata.setdefault('name', skill_md_path.parent.name)
        metadata.setdefault('version', '1.0.0')
        metadata.setdefault('description', '')
        metadata.setdefault('author', 'Unknown')
        
        return metadata
    
    def _build_skill_metadata(self, metadata: Dict, skill_dir: Path, layer: str, priority: int) -> SkillMetadata:
        """构建技能元数据对象"""
        return SkillMetadata(
            name=metadata.get('name', skill_dir.name),
            version=metadata.get('version', '1.0.0'),
            description=metadata.get('description', ''),
            author=metadata.get('author', 'Unknown'),
            created=self._parse_datetime(metadata.get('created')),
            updated=self._parse_datetime(metadata.get('updated')),
            tags=metadata.get('tags', []),
            category=metadata.get('category', 'uncategorized'),
            triggers=metadata.get('triggers', []),
            path=str(skill_dir),
            layer=layer,
            priority=priority,
            mtime=skill_dir.stat().st_mtime
        )
    
    def _parse_datetime(self, value) -> datetime:
        """解析日期时间字符串"""
        if not value:
            return datetime.now()
        
        formats = [
            '%Y-%m-%d',
            '%Y-%m-%dT%H:%M:%S',
            '%Y-%m-%dT%H:%M:%SZ',
            '%Y-%m-%dT%H:%M:%S%z'
        ]
        
        for fmt in formats:
            try:
                return datetime.strptime(value, fmt)
            except ValueError:
                continue
        
        return datetime.now()
```

### 2.3 增量检测机制

**基于 mtime 的增量检测**：
```python
class IncrementalDetector:
    """增量检测器"""
    
    def __init__(self, index_file: str):
        self.index_file = Path(index_file)
        self.previous_index: Dict[str, float] = {}  # skill_path -> mtime
        self._load_index()
    
    def _load_index(self):
        """加载之前的索引"""
        if self.index_file.exists():
            try:
                with open(self.index_file, 'r', encoding='utf-8') as f:
                    self.previous_index = json.load(f)
            except Exception:
                self.previous_index = {}
    
    def _save_index(self):
        """保存索引"""
        self.index_file.parent.mkdir(parents=True, exist_ok=True)
        with open(self.index_file, 'w', encoding='utf-8') as f:
            json.dump(self.previous_index, f, indent=2)
    
    def detect_changes(self, skills: List[SkillMetadata]) -> Dict[str, List[SkillMetadata]]:
        """
        检测技能变化
        
        Returns:
            {
                'new': [新技能列表],
                'modified': [修改的技能列表],
                'unchanged': [未变化的技能列表],
                'deleted': [已删除的技能路径列表]
            }
        """
        current_skills = {s.path: s for s in skills}
        current_paths = set(current_skills.keys())
        previous_paths = set(self.previous_index.keys())
        
        result = {
            'new': [],
            'modified': [],
            'unchanged': [],
            'deleted': []
        }
        
        # 检测新增和修改
        for path, skill in current_skills.items():
            if path not in previous_paths:
                result['new'].append(skill)
            elif skill.mtime > self.previous_index[path]:
                result['modified'].append(skill)
            else:
                result['unchanged'].append(skill)
        
        # 检测删除
        for path in previous_paths:
            if path not in current_paths:
                result['deleted'].append(path)
        
        # 更新索引
        self.previous_index = {path: skill.mtime for path, skill in current_skills.items()}
        self._save_index()
        
        return result
```

### 3.2 注册表实现

```python
class SkillRegistry:
    """技能注册表"""
    
    def __init__(self, index_file: str = "~/.config/opencode/skill_index.json"):
        self.index_file = Path(index_file).expanduser()
        self.skills: Dict[str, SkillMetadata] = {}  # name -> metadata
        self.skills_by_layer: Dict[str, Dict[str, SkillMetadata]] = {
            'workspace': {},
            'local': {},
            'global': {}
        }
        self.scanner = SkillScanner()
        self.detector = IncrementalDetector(str(self.index_file))
        self._initialized = False
    
    def initialize(self, config: Dict):
        """
        初始化注册表，加载所有层的技能
        
        Args:
            config: 配置字典
                {
                    'workspace_dir': '/path/to/workspace/skills',
                    'local_dir': '~/.config/opencode/skills',
                    'global_dirs': ['/usr/local/share/opencode/skills']
                }
        """
        if self._initialized:
            return
        
        # 定义各层扫描配置
        scan_configs = [
            {'dir': config.get('workspace_dir'), 'layer': 'workspace', 'priority': 3},
            {'dir': config.get('local_dir', '~/.config/opencode/skills'), 'layer': 'local', 'priority': 2},
        ]
        
        # 添加全局目录
        for global_dir in config.get('global_dirs', []):
            scan_configs.append({
                'dir': global_dir,
                'layer': 'global',
                'priority': 1
            })
        
        # 扫描各层技能
        all_skills = []
        for cfg in scan_configs:
            if not cfg['dir']:
                continue
            skills = self.scanner.scan_directory(cfg['dir'], cfg['layer'], cfg['priority'])
            all_skills.extend(skills)
        
        # 增量检测
        changes = self.detector.detect_changes(all_skills)
        
        # 构建注册表（按优先级顺序，高优先级覆盖低优先级）
        for skill in sorted(all_skills, key=lambda s: s.priority):
            self.skills[skill.name] = skill
            self.skills_by_layer[skill.layer][skill.name] = skill
        
        # 记录日志
        print(f"技能注册表初始化完成:")
        print(f"  - 新增技能：{len(changes['new'])}")
        print(f"  - 修改技能：{len(changes['modified'])}")
        print(f"  - 删除技能：{len(changes['deleted'])}")
        print(f"  - 总技能数：{len(self.skills)}")
        
        self._initialized = True
    
    def get_skill(self, name: str, layer: Optional[str] = None) -> Optional[SkillMetadata]:
        """
        获取技能元数据
        
        Args:
            name: 技能名称
            layer: 可选，指定层级
        
        Returns:
            技能元数据，不存在则返回 None
        """
        if layer:
            return self.skills_by_layer.get(layer, {}).get(name)
        return self.skills.get(name)
    
    def list_skills(self, layer: Optional[str] = None, category: Optional[str] = None,
                    tags: Optional[List[str]] = None) -> List[SkillMetadata]:
        """
        列出技能（支持过滤）
        
        Args:
            layer: 按层级过滤
            category: 按分类过滤
            tags: 按标签过滤（AND 逻辑）
        
        Returns:
            技能元数据列表
        """
        skills = []
        
        if layer:
            skills = list(self.skills_by_layer.get(layer, {}).values())
        else:
            skills = list(self.skills.values())
        
        # 按分类过滤
        if category:
            skills = [s for s in skills if s.category == category]
        
        # 按标签过滤（AND 逻辑）
        if tags:
            skills = [s for s in skills if all(tag in s.tags for tag in tags)]
        
        # 按优先级排序
        return sorted(skills, key=lambda s: s.priority, reverse=True)
    
    def search_skills(self, query: str) -> List[SkillMetadata]:
        """
        搜索技能（模糊匹配）
        
        Args:
            query: 搜索关键词
        
        Returns:
            匹配的技能列表
        """
        query_lower = query.lower()
        results = []
```

## 4. 技能冲突解决

### 4.1 冲突场景

| 冲突类型 | 描述 | 示例 |
|---------|------|------|
| **命名冲突** | 不同层存在同名技能 | 工作区和本地都有 `code-review` |
| **版本冲突** | 同名技能版本不兼容 | 工作区 v2.0 vs 本地 v1.0 |
| **依赖冲突** | 技能依赖不同版本的第三方库 | Skill-A 依赖 lib v1.0, Skill-B 依赖 lib v2.0 |
| **触发器冲突** | 多个技能响应相同触发词 | `code-review` 和 `security-check` 都响应 "检查代码" |

### 4.2 冲突解决策略

```python
class ConflictResolver:
    """技能冲突解决器"""
    
    def resolve_naming_conflicts(self, skills: List[SkillMetadata]) -> Dict[str, SkillMetadata]:
        """
        解决命名冲突（高优先级覆盖低优先级）
        
        Args:
            skills: 所有技能列表
        
        Returns:
            解决冲突后的技能字典
        """
        registry = {}
        
        # 按优先级排序（高优先级在后，会覆盖低优先级）
        sorted_skills = sorted(skills, key=lambda s: s.priority)
        
        for skill in sorted_skills:
            if skill.name in registry:
                existing = registry[skill.name]
                print(f"技能命名冲突：{skill.name}")
                print(f"  - 已存在：{existing.path} (priority={existing.priority})")
                print(f"  - 新技能：{skill.path} (priority={skill.priority})")
                print(f"  → 使用高优先级技能")
            
            registry[skill.name] = skill
        
        return registry
    
    def check_version_compatibility(self, skill: SkillMetadata, required_version: str) -> bool:
        """
        检查版本兼容性（semver）
        
        Args:
            skill: 技能元数据
            required_version: 要求的版本（如 "^1.0.0"）
        
        Returns:
            是否兼容
        """
        import re
        
        current_version = skill.version
        
        # 解析版本要求
        match = re.match(r'^([\^~>=<]*)(\d+\.\d+\.\d+)', required_version)
        if not match:
            return False
        
        operator, required = match.groups()
        
        # 简单版本比较
        current_parts = [int(x) for x in current_version.split('.')]
        required_parts = [int(x) for x in required.split('.')]
        
        if operator == '^':  # 兼容主版本
            return current_parts[0] == required_parts[0] and current_parts >= required_parts
        elif operator == '~':  # 兼容次版本
            return (current_parts[0] == required_parts[0] and
                    current_parts[1] == required_parts[1] and
                    current_parts >= required_parts)
        elif operator == '>=':
            return current_parts >= required_parts
        elif operator == '=':
            return current_parts == required_parts
        
        return False
    
    def detect_trigger_conflicts(self, skills: List[SkillMetadata]) -> List[Dict]:
        """
        检测触发器冲突
        
        Returns:
            冲突列表
        """
        trigger_map = {}
        conflicts = []
        
        for skill in skills:
            for trigger in skill.triggers:
                if trigger not in trigger_map:
                    trigger_map[trigger] = []
                trigger_map[trigger].append(skill)
        
        for trigger, conflicting_skills in trigger_map.items():
            if len(conflicting_skills) > 1:
                conflicts.append({
                    'trigger': trigger,
                    'skills': conflicting_skills,
                    'resolution': '使用高优先级技能'
                })
        
        return conflicts
```

## 5. 技能加载性能优化

### 5.1 懒加载策略

**核心思想**：启动时只加载元数据，执行时才加载完整技能定义。

```python
class LazySkillLoader:
    """懒加载技能加载器"""
    
    def __init__(self, registry: SkillRegistry):
        self.registry = registry
        self.loaded_skills: Dict[str, Dict] = {}  # 缓存已加载的完整技能
    
    def get_skill_metadata(self, name: str) -> Optional[SkillMetadata]:
        """快速获取元数据（不加载完整技能）"""
        return self.registry.get_skill(name)
    
    def get_skill_full(self, name: str) -> Optional[Dict]:
        """
        获取完整技能定义（懒加载）
        
        Args:
            name: 技能名称
        
        Returns:
            完整技能定义（包含 reference/scripts 等）
        """
        # 检查缓存
        if name in self.loaded_skills:
            return self.loaded_skills[name]
        
        # 获取元数据
        metadata = self.registry.get_skill(name)
        if not metadata:
            return None
        
        # 加载完整技能
        skill_full = self._load_skill_content(metadata)
        
        # 缓存
        self.loaded_skills[name] = skill_full
        
        return skill_full
    
    def _load_skill_content(self, metadata: SkillMetadata) -> Dict:
        """加载技能完整内容"""
        skill_path = Path(metadata.path)
        
        content = {
            'metadata': metadata,
            'definition': None,
            'reference': {},
            'scripts': {},
            'examples': {}
        }
        
        # 加载 SKILL.md
        skill_md = skill_path / 'SKILL.md'
        if skill_md.exists():
            with open(skill_md, 'r', encoding='utf-8') as f:
                content['definition'] = f.read()
        
        # 加载 reference（懒加载，按需）
        ref_dir = skill_path / 'reference'
        if ref_dir.exists():
            for ref_file in ref_dir.glob('*.md'):
                # 这里可以先不加载，等需要时再加载
                content['reference'][ref_file.stem] = ref_file
        
        # 加载 scripts
        scripts_dir = skill_path / 'scripts'
        if scripts_dir.exists():
            for script_file in scripts_dir.glob('*.py'):
                with open(script_file, 'r', encoding='utf-8') as f:
                    content['scripts'][script_file.stem] = f.read()
        
        return content
    
    def clear_cache(self, name: Optional[str] = None):
        """清除缓存"""
        if name:
            self.loaded_skills.pop(name, None)
        else:
            self.loaded_skills.clear()
```

### 5.3 缓存失效策略

```python
class CacheManager:
    """缓存管理器"""
    
    def __init__(self, registry: SkillRegistry, lazy_loader: LazySkillLoader):
        self.registry = registry
        self.lazy_loader = lazy_loader
        self.cache_ttl = 3600  # 1 小时
    
    def is_cache_valid(self, skill_name: str) -> bool:
        """检查缓存是否有效"""
        metadata = self.registry.get_skill(skill_name)
        if not metadata:
            return False
        
        # 检查文件修改时间
        skill_path = Path(metadata.path)
        current_mtime = skill_path.stat().st_mtime
        
        return current_mtime == metadata.mtime
    
    def invalidate_if_needed(self):
        """失效过期的缓存"""
        for skill_name in list(self.lazy_loader.loaded_skills.keys()):
            if not self.is_cache_valid(skill_name):
                self.lazy_loader.clear_cache(skill_name)
                print(f"缓存失效：{skill_name}")
```

```python
from pathlib import Path
from typing import Dict, List, Optional
import json

class SkillLoader:
    """
    完整技能加载器
    
    整合扫描器、注册表、懒加载、缓存管理
    """
    
    def __init__(self, config: Dict):
        self.config = config
        self.registry = SkillRegistry()
        self.lazy_loader = LazySkillLoader(self.registry)
        self.cache_manager = CacheManager(self.registry, self.lazy_loader)
        self.preloader = Preloader(self.lazy_loader)
    
    def initialize(self, preload: bool = True):
        """
        初始化技能加载器
        
        Args:
            preload: 是否预加载常用技能
        """
        # 初始化注册表
        self.registry.initialize({
            'workspace_dir': self.config.get('workspace_dir'),
            'local_dir': self.config.get('local_dir', '~/.config/opencode/skills'),
            'global_dirs': self.config.get('global_dirs', [])
        })
        
        # 预加载
        if preload:
            self.preloader.preload()
    
    def get_skill(self, name: str, full: bool = False):
        """
        获取技能
        
        Args:
            name: 技能名称
            full: 是否返回完整定义
        
        Returns:
            技能元数据或完整定义
        """
        if full:
            return self.lazy_loader.get_skill_full(name)
        return self.lazy_loader.get_skill_metadata(name)
    
    def list_skills(self, **kwargs):
        """列出技能"""
        return self.registry.list_skills(**kwargs)
    
    def search_skills(self, query: str):
        """搜索技能"""
        return self.registry.search_skills(query)
    
    def refresh(self):
        """刷新技能注册表"""
        self.cache_manager.invalidate_if_needed()
        self.registry.refresh()
```
